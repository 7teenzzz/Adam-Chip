    ```bash
    echo "Using discard_window_ms = $DISCARD_MS"

    PYTHONPATH=System ./.venv/bin/python - <<PY
    import json
    from pathlib import Path
    events = [json.loads(l) for l in Path("data/adam/events.jsonl").read_text().splitlines() if l.strip()]
    discard_ms = $DISCARD_MS

    # Берём только события за последние 30 мин (по unix ts), чтобы не учитывать pre-run артефакты
    import time
    cutoff = float(open("/tmp/29-05-start-ts.txt").read().strip())
    events = [e for e in events if e.get("ts", 0) >= cutoff]
    print(f"Events analyzed since {cutoff}: {len(events)}")

    violations = []
    pairs = 0
    i = 0
    while i < len(events):
        if events[i].get("event") == "tts_started":
            t_start = events[i].get("ts", 0)
            target = None
            t_end = None
            for j in range(i+1, len(events)):
                if events[j].get("event") == "tts_finished":
                    t_end = events[j].get("ts", 0)
                    target = events[j].get("payload", {}).get("target")
                    break
            if t_end is None:
                i += 1
                continue
            pairs += 1
            window_end_ts = t_end + (discard_ms / 1000.0)
            for k in range(i+1, len(events)):
                if events[k].get("event") == "asr_result":
                    tk = events[k].get("ts", 0)
                    if t_start <= tk <= window_end_ts:
                        violations.append({"asr_ts": tk, "tts_start": t_start, "tts_end": t_end, "target": target})
            i = i + 1
        else:
            i += 1
    print(f"Pairs analyzed: {pairs}")
    print(f"Violations: {len(violations)}")
    for v in violations[:10]:
        print(v)
    PY
    ```
